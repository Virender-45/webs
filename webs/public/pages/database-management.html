Query Examples

-- Find top 5 students by average score with their course SELECT s.student_name, c.course_title, AVG(e.score) AS avg_score FROM students s JOIN enrollments e ON s.id = e.student_id JOIN courses c ON e.course_id = c.id GROUP BY s.id, c.id ORDER BY avg_score DESC LIMIT 5;
